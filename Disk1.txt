#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, head;

    printf("Enter number of requests: ");
    scanf("%d", &n);

    int req[n];

    printf("Enter the requests:\n");
    for(int i = 0; i < n; i++) {
        scanf("%d", &req[i]);
    }

    printf("Enter initial head position: ");
    scanf("%d", &head);

    int total_seek = 0;

    printf("\nSeek Sequence:\n");

    // FCFS logic
    for(int i = 0; i < n; i++) {
        printf("%d -> ", req[i]);

        total_seek += abs(head - req[i]); // calculate seek time
        head = req[i]; // move head
    }

    printf("\n\nTotal Seek Time = %d\n", total_seek);

    return 0;
}

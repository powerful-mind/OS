#include <stdio.h>
#include <limits.h>

struct Process {
    int pid;
    int arrivalTime;
    int burstTime;
    int remainingTime;
    int completionTime;
    int waitingTime;
    int turnaroundTime;
};

int main() {
    int n;
    printf("Enter number of processes: ");
    scanf("%d", &n);

    struct Process p[n];

    // Input
    for (int i = 0; i < n; i++) {
        p[i].pid = i + 1;
        printf("Enter Arrival Time for Process %d: ", p[i].pid);
        scanf("%d", &p[i].arrivalTime);
        printf("Enter Burst Time for Process %d: ", p[i].pid);
        scanf("%d", &p[i].burstTime);
        p[i].remainingTime = p[i].burstTime;
    }

    int completed = 0, currentTime = 0;

    // Gantt chart (more slots due to preemption)
    int ganttTime[1000];
    char ganttLabel[1000][10];
    int g = 0;

    while (completed < n) {
        int idx = -1;
        int minRT = INT_MAX;

        // Find process with minimum remaining time
        for (int i = 0; i < n; i++) {
            if (p[i].arrivalTime <= currentTime && p[i].remainingTime > 0) {
                if (p[i].remainingTime < minRT) {
                    minRT = p[i].remainingTime;
                    idx = i;
                }
            }
        }

        if (idx == -1) {
            // CPU IDLE
            sprintf(ganttLabel[g], "IDLE");
            ganttTime[g] = currentTime;
            currentTime++;
            ganttTime[g + 1] = currentTime;
            g++;
        } else {
            // Execute for 1 unit
            sprintf(ganttLabel[g], "P%d", p[idx].pid);
            ganttTime[g] = currentTime;

            p[idx].remainingTime--;
            currentTime++;

            ganttTime[g + 1] = currentTime;
            g++;

            // If completed
            if (p[idx].remainingTime == 0) {
                completed++;
                p[idx].completionTime = currentTime;
                p[idx].turnaroundTime = p[idx].completionTime - p[idx].arrivalTime;
                p[idx].waitingTime = p[idx].turnaroundTime - p[idx].burstTime;
            }
        }
    }

    float totalWT = 0, totalTAT = 0;

    printf("\nPID\tAT\tBT\tCT\tWT\tTAT\n");
    for (int i = 0; i < n; i++) {
        printf("%d\t%d\t%d\t%d\t%d\t%d\n",
               p[i].pid,
               p[i].arrivalTime,
               p[i].burstTime,
               p[i].completionTime,
               p[i].waitingTime,
               p[i].turnaroundTime);

        totalWT += p[i].waitingTime;
        totalTAT += p[i].turnaroundTime;
    }

    // Gantt Chart
    printf("\nGantt Chart:\n");

    for (int i = 0; i < g; i++) printf("--------");
    printf("-\n|");

    for (int i = 0; i < g; i++) {
        printf(" %-5s |", ganttLabel[i]);
    }

    printf("\n");

    for (int i = 0; i < g; i++) printf("--------");
    printf("-\n");

    for (int i = 0; i <= g; i++) {
        printf("%-8d", ganttTime[i]);
    }

    printf("\n\nAverage Waiting Time: %.2f", totalWT / n);
    printf("\nAverage Turnaround Time: %.2f\n", totalTAT / n);

    return 0;
}

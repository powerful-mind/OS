#include <stdio.h>

int main() {
    int frames = 3;
    int page[] = {7, 0, 1, 2, 0, 3, 0, 4, 2, 3, 0, 3, 2, 1, 2};
    int n = 15;

    int frame[3];
    int i, j, k, faults = 0;

    // Initialize frames
    for(i = 0; i < frames; i++) {
        frame[i] = -1;
    }

    printf("Optimal Page Replacement:\n");

    for(i = 0; i < n; i++) {
        int found = 0;

        // Check HIT
        for(j = 0; j < frames; j++) {
            if(frame[j] == page[i]) {
                found = 1;
                break;
            }
        }

        // PAGE FAULT
        if(!found) {
            int index = -1, farthest = i;

            // Find page to replace
            for(j = 0; j < frames; j++) {
                int next_use = -1;

                // Look ahead in future
                for(k = i + 1; k < n; k++) {
                    if(frame[j] == page[k]) {
                        next_use = k;
                        break;
                    }
                }

                // If page not used again → replace immediately
                if(next_use == -1) {
                    index = j;
                    break;
                }

                // Choose farthest future use
                if(next_use > farthest) {
                    farthest = next_use;
                    index = j;
                }
            }

            // If all pages used again, replace the farthest one
            if(index == -1)
                index = 0;

            frame[index] = page[i];
            faults++;

            printf("Page %d -> ", page[i]);
            for(j = 0; j < frames; j++) {
                if(frame[j] != -1)
                    printf("%d ", frame[j]);
                else
                    printf("- ");
            }
            printf("\n");
        }
    }

    printf("Total Page Faults = %d\n", faults);

    return 0;
}

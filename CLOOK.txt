#include <stdio.h>
#include <stdlib.h>

// Bubble sort
void sort(int a[], int n) {
    for(int i = 0; i < n - 1; i++) {
        for(int j = 0; j < n - i - 1; j++) {
            if(a[j] > a[j + 1]) {
                int t = a[j];
                a[j] = a[j + 1];
                a[j + 1] = t;
            }
        }
    }
}

int main() {
    int n, head, dir;

    printf("Enter number of requests: ");
    scanf("%d", &n);

    int req[n];

    printf("Enter the requests:\n");
    for(int i = 0; i < n; i++)
        scanf("%d", &req[i]);

    printf("Enter head position: ");
    scanf("%d", &head);

    printf("Enter direction (1 = right, 0 = left): ");
    scanf("%d", &dir);

    // Sort requests
    sort(req, n);

    int total_seek = 0;
    int pos = n;

    // Find split position
    for(int i = 0; i < n; i++) {
        if(req[i] >= head) {
            pos = i;
            break;
        }
    }

    printf("\nSeek Sequence:\n");

    if(dir == 1) { // Move right
        // Serve right side
        for(int i = pos; i < n; i++) {
            printf("%d -> ", req[i]);
            total_seek += abs(head - req[i]);
            head = req[i];
        }

        // Jump to first request (not 0)
        if(pos > 0) {
            printf("%d -> ", req[0]);
            total_seek += abs(head - req[0]);
            head = req[0];
        }

        // Continue right
        for(int i = 1; i < pos; i++) {
            printf("%d -> ", req[i]);
            total_seek += abs(head - req[i]);
            head = req[i];
        }
    }
    else { // Move left
        // Serve left side
        for(int i = pos - 1; i >= 0; i--) {
            printf("%d -> ", req[i]);
            total_seek += abs(head - req[i]);
            head = req[i];
        }

        // Jump to last request
        if(pos < n) {
            printf("%d -> ", req[n - 1]);
            total_seek += abs(head - req[n - 1]);
            head = req[n - 1];
        }

        // Continue left
        for(int i = n - 2; i >= pos; i--) {
            printf("%d -> ", req[i]);
            total_seek += abs(head - req[i]);
            head = req[i];
        }
    }

    printf("\n\nTotal Seek Time = %d\n", total_seek);

    return 0;
}

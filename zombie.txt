#include <stdio.h> 
#include <stdlib.h> 
#include <unistd.h> 
#include <sys/wait.h> 
#include <sys/types.h> 
 
// Bubble Sort - used by CHILD process 
void bubbleSort(int arr[], int n) { 
    int i, j, temp; 
    for (i = 0; i < n-1; i++) 
        for (j = 0; j < n-i-1; j++) 
            if (arr[j] > arr[j+1]) { 
                temp = arr[j]; 
                arr[j] = arr[j+1]; 
                arr[j+1] = temp; 
            } 
} 
 
// Selection Sort - used by PARENT process 
void selectionSort(int arr[], int n) { 
    int i, j, minIdx, temp; 
    for (i = 0; i < n-1; i++) { 
        minIdx = i; 
        for (j = i+1; j < n; j++) 
            if (arr[j] < arr[minIdx]) minIdx = j; 
        temp = arr[minIdx]; arr[minIdx] = arr[i]; arr[i] = temp; 
    } 
} 
 
 
void printArray(int arr[], int n) { 
    for (int i = 0; i < n; i++) 
        printf("%d\n", arr[i]); 
} 
 
int main() { 
    int n, i; 
    printf("Enter number of integers: "); 
   printf("Enter number of integers: "); 
    scanf("%d", &n); 
    int arr[n], arrCopy[n]; 
    printf("Enter %d integers: ", n); 
    for (i = 0; i < n; i++) { 
        scanf("%d", &arr[i]); 
        arrCopy[i] = arr[i]; 
    } 
 
    printf("Original Array: "); 
    printArray(arr, n); 
 
    pid_t p = fork(); 
 
    if (p < 0) { 
        perror("fork failed"); 
        exit(1); 
    } else if (p == 0) { 
        // ── CHILD PROCESS ── 
        printf("\n[CHILD] PID=%d, PPID=%d\n", getpid(), getppid()); 
        printf("[CHILD] Sorting using Bubble Sort...\n"); 
        bubbleSort(arrCopy, n); 
        printf("[CHILD] Sorted Array: "); 
        printArray(arrCopy, n); 
        printf("[CHILD] Done. Exiting.\n"); 
        exit(0); 
    } else { 
        // ── PARENT PROCESS ── 
        printf("\n[PARENT] PID=%d, PPID=%d, Child PID=%d\n", 
                getpid(), getppid(), p); 
 
        printf("[PARENT] Sorting using Selection Sort...\n"); 
        selectionSort(arr, n); 
        printf("[PARENT] Sorted Array: "); 
        printArray(arr, n); 
        printf("[PARENT] Waiting for child...\n"); 
        wait(NULL); 
        printf("[PARENT] Child finished. Exiting.\n"); 
} 
    return 0; 
}

int main() { 
    pid_t p = fork(); 
    if (p == 0) { 
        printf("[CHILD] PID=%d - Exiting now (becoming ZOMBIE)\n", getpid()); 
        exit(0);          // child exits immediately 
    } else { 
        printf("[PARENT] PID=%d, Child PID=%d\n", getpid(), p); 
        printf("[PARENT] Sleeping 30s... Check zombie in another terminal:\n"); 
        printf("         Run: ps aux | grep Z\n"); 
        sleep(30);        // parent does NOT call wait() - child is zombie! 
        wait(NULL);       // now reap the zombie 
        printf("[PARENT] Zombie reaped. Done.\n"); 
    } 
    return 0; 
}

#include <stdio.h> 
#include <stdlib.h> 
#include <unistd.h> 
 
int main() { 
    pid_t p = fork(); 
 
    if (p < 0) { 
        perror("Fork failed"); 
        exit(1); 
    } 
 
    if (p > 0) { 
        // Parent Process 
        printf("[PARENT] PID=%d - Exiting now!\n", getpid()); 
        exit(0);  
    } else { 
        // Child Process 
        // Displaying PID and PPID before the parent exits 
        printf("[CHILD]  PID=%d, Initial PPID=%d\n", getpid(), getppid()); 
         
        printf("[CHILD]  Sleeping for 3 seconds to become an orphan...\n"); 
        sleep(3);  
         
        // Displaying PID and the NEW PPID after adoption 
        // Note: Added %d for the Child's PID here 
        printf("[CHILD]  After sleep: My PID=%d, My NEW PPID=%d (Adopted!)\n",  
               getpid(), getppid()); 
    } 
 
    return 0; 
}
